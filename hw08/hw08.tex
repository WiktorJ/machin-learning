\section{Problem 1}
With linear activation functions whole network reduces to
linear transformation.
\begin{equation}
    f(x, W) = W_k(W_{k-1}...(W_0x)) = W_kW_{k-1}...W_0x = W'x
\end{equation}
Thus we would be limited only to linear functions.\\
With appropriate non linear activation functions, neural net is able to 
approximate any (differentiable) function. This property allows the net 
to find very complicated and complex patterns in data and create non linear
mapping between input and output.
\section{Problem 2}
\begin{equation}
    \sigma(x) = \frac{e^x}{e^x + 1} = 
    \frac{e^x - (\frac{1}{2}e^x + \frac{1}{2}) + (\frac{1}{2}e^x + \frac{1}{2})}{e^x + 1} = 
    \frac{\frac{1}{2}e^x - \frac{1}{2}}{e^x + 1} + \frac{\frac{1}{2}(e^x + 1)}{e^x + 1} = 
    \frac{1}{2} \frac{e^x - 1}{e^x + 1} + \frac{1}{2} = \frac{1}{2}tanh(\frac{1}{2}x) + \frac{1}{2}
\end{equation}  
From above we see that we can compute $\sigma(x)$ value using $tanh(x)$.\\\\
So instead of using $\sigma(w^Tx)$ as activation function, we can use:
\begin{equation}
    g(w,x) = \frac{1}{2}tanh(\frac{1}{2}w^Tx) + \frac{1}{2}
\end{equation}
To approximate the same function using a neural network.
\section{Problem 3}
\begin{equation}
    tanh(x) = \frac{sinh(x)}{cosh(x)}
\end{equation}
\begin{equation}
    sinh(x) = \frac{e^x - e^{-x}}{2}
\end{equation}
\begin{equation}
    cosh(x) = \frac{e^x + e^{-x}}{2} 
\end{equation}
\begin{equation}
    sinh'(x) = (\frac{e^x - e^{-x}}{2})' = \frac{1}{2}(e^x - e^{-x})' =
    \frac{1}{2}(e^x - \frac{-1}{e^{2x}} * e^x) =
    \frac{1}{2}(e^x + e^{-x}) = cosh(x)
\end{equation}
\begin{equation}
    cosh'(x) = (\frac{e^x + e^{-x}}{2})' = \frac{1}{2}(e^x + e^{-x})' =
    \frac{1}{2}(e^x + \frac{-1}{e^{2x}} * e^x) =
    \frac{1}{2}(e^x - e^{-x}) = sinh(x)
\end{equation}
\begin{equation}
    tanh'(x) = (\frac{sinh(x)}{cosh(x)})' = 
    \frac{cosh(x)sinh'(x) - sinh(x) cosh'(x) }{cosh^2(x)} =
    \frac{cosh^2(x) - sinh^2(x)}{cosh^2(x)} = 1 - tanh^2(x)
\end{equation}

\section{Problem 6}
\begin{equation}
    \sigma(x) = \frac{e^x}{e^x + 1}
\end{equation}  
\begin{equation}  
    1 - \sigma(x) = 1 - \frac{e^x}{e^x + 1} = 
    \frac{e^x + 1}{e^x + 1} - \frac{e^x}{e^x + 1} = \frac{1}{e^x + 1}
\end{equation}
\begin{equation}
    \begin{gathered}
        f(x) = -(y log(\sigma(x)) + (1-y)log(1-\sigma(x))) = \\
        -(y log(\frac{e^x}{e^x + 1}) + (1-y)log(\frac{1}{e^x + 1})) = \\
        -(y (log({e^x}) - log({e^x + 1})) + (1-y)(log({1}) - log({e^x + 1}))) = \\
        -(yx - y log({e^x + 1}) + (1-y)(- log({e^x + 1}))) =  \\
        -(yx - y log({e^x + 1}) - log({e^x + 1}) + y log({e^x + 1}) = \\
        -yx + log({e^x + 1}) = \\
        -yx + log(e^x (1 + e^{-x})) = \\
        -yx + log(e^x) + log(1 + e^{-x}) = \\
        -yx + x + log(1 + e^{-x})
    \end{gathered}    
\end{equation}
Now we can also define $f$ in this way.
\[ f(x) =  \begin{cases} 
    -yx + log({e^x + 1}) & x < 0 \\
    -yx + x + log(1 + e^{-x}) & x \geq 0  
 \end{cases}
\]
As it was shown above that both forms are equivalent.\\
From here we can also reduce this conditional function into 
one equation:
\begin{equation}
    f(x) = -yx + max(x, 0) + log(1 + e^{-|x|})
\end{equation}
